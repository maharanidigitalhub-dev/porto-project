// ── Nav scroll ──────────────────────────────────────────────────────────────

// ── Reveal on scroll ────────────────────────────────────────────────────────
const revealObserver = new IntersectionObserver((entries) => {
  entries.forEach(e => {
    if (e.isIntersecting) {
      e.target.classList.add('visible');
      revealObserver.unobserve(e.target);
    }
  });
}, { threshold: 0.12, rootMargin: '0px 0px -40px 0px' });

document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));

// ── Counter animation ───────────────────────────────────────────────────────
const counterObserver = new IntersectionObserver((entries) => {
  entries.forEach(e => {
    if (e.isIntersecting) {
      const el = e.target;
      const target = parseInt(el.dataset.target);
      const suffix = el.dataset.suffix || '';
      if (!target) return;
      let start = 0;
      const duration = 1600;
      const startTime = performance.now();
      function update(now) {
        const elapsed = now - startTime;
        const progress = Math.min(elapsed / duration, 1);
        const ease = 1 - Math.pow(1 - progress, 3);
        start = Math.round(ease * target);
        el.textContent = start + suffix;
        if (progress < 1) requestAnimationFrame(update);
      }
      requestAnimationFrame(update);
      counterObserver.unobserve(el);
    }
  });
}, { threshold: 0.5 });

document.querySelectorAll('[data-target]').forEach(el => counterObserver.observe(el));

// ── Accordion principle ─────────────────────────────────────────────────────
function togglePrinciple(btn) {
  const item = btn.closest('.principle-item');
  const isActive = item.classList.contains('active');
  document.querySelectorAll('.principle-item').forEach(i => i.classList.remove('active'));
  if (!isActive) item.classList.add('active');
}

// ── Case study tabs ──────────────────────────────────────────────────────────

  document.querySelectorAll('.case-panel').forEach(p => p.classList.remove('active'));
  btn.classList.add('active');
  document.getElementById(id).classList.add('active');
}
